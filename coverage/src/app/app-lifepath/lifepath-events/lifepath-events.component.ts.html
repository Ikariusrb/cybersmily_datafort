import { DiceService } from './../../shared/services/dice/dice.service';
import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { LifepathEvent } from './../../shared/models/lifepath';
 
@Component({
  selector: 'cs-lifepath-events',
  templateUrl: './lifepath-events.component.html',
  styleUrls: ['./lifepath-events.component.css']
})
export class LifepathEventsComponent implements OnInit {
  @Input()
  chartData: any;
 
  @Input()
  lifepathSource: string;
 
  @Output()
  generateEvents: EventEmitter<LifepathEvent[]> = new EventEmitter<LifepathEvent[]>();
 
  levents: LifepathEvent[] = new Array<LifepathEvent>();
  hasEventful = false;
  Years: number;
  InputYear: string;
 
  constructor(private dice: DiceService) { }
 
  ngOnInit() {
  }
 
 
  /**
   * generate the list of life events, starting from age 16.
   *
   * @memberof LifepathEventsComponent
   */
  generateLifeEvents() {
    const lifeEvents: any[] = (typeof (this.chartData.ChartRoll[this.lifepathSource]) === 'undefined') ?
        this.chartData.ChartRoll['CP2020'] : this.chartData.ChartRoll[this.lifepathSource];
    this.levents = [];
    const startAge = 16;
 
    this.Years = this.getNumOfYears(this.InputYear);
    this.InputYear = this.Years.toString();
    // for each year roll for an event
    for ( let j = 0; j < this.Years; j++) {
        const event = this.generateEvent(lifeEvents);
        this.levents.push( { age: (startAge + j), event: event });
    }
    this.emitLifeEvents(this.levents);
  }
 
  /**
   *Generate an event from the input chart.
   * @param {*} chart - chart to find the event type to generate.
   * @returns {string} - the event generated.
   * @memberof LifepathEventsComponent
   */
  generateEvent(chart): string {
    // if hasEventful is true, then ignore the rolls for 'nothing happened.'
    const die = (this.hasEventful) ? 7 : 9;
    // determine the event type
    const mainRoll = '' + this.dice.generateNumber(0, die);
    const mainEvent = chart[mainRoll];
    let eventDesc = `Nothing happened that year.`;
    // check to see if there is a chart for the main event,
    // if not then it means nothing happened.
    if ( typeof(mainEvent.chart) !== 'undefined') {
        const roll = this.dice.generateNumber(0, (mainEvent.chart.length - 1));
        const eventType = mainEvent.chart[roll];
        eventDesc = this[eventType](this.lifepathSource);
    }
    return eventDesc;
  }
 
  /**
   * get the number of years as number or generate a random number.
   * @param {*} input
   * @returns {number}
   * @memberof LifepathEventsComponent
   */
  getNumOfYears(input: string): number {
    let years = parseInt(input, 0);
    if (isNaN(years) || years < 1) {
        years = this.dice.generateNumber(2, 12);
   }
   return years;
  }
 
  // Big Wins
  GetWin(version) {
      const c = 'Big Win: ' + this.dice.rollOnChart(this.chartData.Wins, version);
      return c;
  }
 
  // Big Problems
  GetDisaster(version) {
      let c = 'Big Lose: ' + this.dice.rollOnChart(this.chartData.Disasters.Event, version);
      c += ' What are you going to do: ' + this.dice.rollOnChart(this.chartData.Disasters.Response, version);
      return c;
  }
 
  // Friends
  GetFriend(version) {
      const isMale = ((this.dice.generateNumber(1, 10) % 2) === 0);
      let s = 'Made a Friend: ' + ((isMale) ? 'He\'s ' : 'She\'s ') + this.dice.rollOnChart(this.chartData.Friends, version);
      if (s.indexOf('{0}') > -1) {
        s = s.replace('{0}', ((isMale) ? 'brother' : 'sister'));
      }
      return s;
  }
 
  // Enemies
  GetEnemy(version) {
      const isMale = ((this.dice.generateNumber(1, 10) % 2) === 0);
      let s = 'Made an Enemy: ' + ((isMale) ? 'He\'s ' : 'She\'s ');
      s += this.dice.rollOnChart(this.chartData.Enemies.who, version);
      s += ' Emnity started due to: ' + this.dice.rollOnChart(this.chartData.Enemies.cause, version);
      s += ' ' + this.dice.rollOnChart(this.chartData.Enemies.fracked, version);
      s += ' Next time you meet, reaction is: ' + this.dice.rollOnChart(this.chartData.Enemies.reaction, version);
      s += ' Their backup is ' + this.dice.rollOnChart(this.chartData.Enemies.resource, version);
      if (!isMale && s.indexOf('his') > -1) { s = s.replace(/his/gi, 'her'); }
      if (!isMale && s.indexOf('him') > -1) { s = s.replace(/him/gi, 'her'); }
      if (!isMale && s.indexOf('him') > -1) { s = s.replace('he\'s', 'she\'s'); }
      return s;
  }
 
  // Romantic Involvement
  GetRomance(version) {
      let s = 'Romantic Involvement: ' + this.dice.rollOnChart(this.chartData.RomanticInvolvement.Type, version);
      if (version === 'ProtectServe') {
          s += ' ' + this.dice.rollOnChart(this.chartData.RomanticInvolvement.Type, version);
      }
      if (s.indexOf('Tragic') < 0 && s.indexOf('Problem') < 0) {
        return s;
      }
      if (s.indexOf('Tragic') > -1) {
        s += ' ' + this.dice.rollOnChart(this.chartData.RomanticInvolvement.Tragic, version);
      } else {
        s += ' ' + this.dice.rollOnChart(this.chartData.RomanticInvolvement.Problems, version);
      }
      s += ' ' + this.dice.rollOnChart(this.chartData.RomanticInvolvement.Feeling, version);
      return s;
  }
 
  emitLifeEvents(events: LifepathEvent[]) {
    this.generateEvents.emit(events);
  }
}
