import { MaxMetalVehicle, VehicleType } from './../models/maxmetal';
import { Injectable } from '@angular/core';
import { BehaviorSubject } from 'rxjs';
 
@Injectable()
export class MaxmetalService {
  private vehicle = new BehaviorSubject<MaxMetalVehicle>(new MaxMetalVehicle());
  cast = this.vehicle.asObservable();

  constructor() {

  /**
   * sets the current vehicles type. This will also set up the initial
   * current vehicle's properties.
   * @param {VehicleType} type
   * @memberof MaxmetalService
   */
  setVehicleType(type: VehicleType) {
    this.currVehicle = new MaxMetalVehicle();
    this.currVehicle.type = type;
    // set the sdp properties.
    this.currVehicle.sdp = {min: type.sdp.min, max: type.sdp.max, curr: type.sdp.min, base: type.sdp.min};
    // set the adj sdp properties
    this.calculateAdjSDP();
    // set the armor properties
    this.calculateSP();
    // set the top speed
    this.calculateSpeed();
    // set the acceleration
    this.currVehicle.accelerate = {min: type.acc, max: type.acc * 2, base: type.acc, curr: type.acc, cost: 1};
    // set the deceleration
    this.currVehicle.decelerate.min = type.dec;
    this.currVehicle.decelerate.max = (type.isAirVehicle) ? Math.round(type.dec * 1.5) : Math.round(type.dec * 2);
    this.currVehicle.decelerate.base = type.dec;
    this.currVehicle.decelerate.curr = type.dec;
    // set the range
    this.currVehicle.range = {min: Math.round(type.range * 0.01), max: Math.round(type.range * 4.33), base: type.range, curr: type.range};
    // set the initial spaces
    this.currVehicle.spaces = {min: type.spaces.min, max: type.spaces.max, base: type.spaces.min, curr: type.spaces.min};
    // set the initial mass
    this.calculateMass();
    this.calculateCost();
 
    this.vehicle.next(this.currVehicle);
  }
 
  /**
   * change the value of the current vehicle's SDP and then
   * calculates all the other properties associated with that change.
   * @param {number} value
   * @memberof MaxmetalService
   */
  changeSDP(value: number) {
      this.currVehicle.sdp.base += value;
      this.currVehicle.sdp.curr += value;
      this.calculateAdjSDP();
      this.calculateSP();
      this.calculateMass();
      this.calculateCost();
      this.calculateSpace();
  }
 
 
  /**
   * MAX METAL pg 14
   * Added Strucutre: Vehicles with superior construction (heavier materials, roll
   * cages, etc.) may have up to 25% more SDP than the maximum for their vehicle
   * type. Each extra point of SDP costs double the normal SDP cost.
   * Weaker Structure: Cheap vehciles may have only half of their required SDP. This
   * makes them more liable to combat damage and has the additional detriment of making
   * them break down often during play.
   * @param {number} value
   * @memberof MaxmetalService
   */
  changeExtraSDP(value: number) {
    this.currVehicle.sdpAdj.curr += value;
    this.currVehicle.sdp.curr += value;
    this.calculateSP();
    this.calculateMass();
    this.calculateCost();
    this.calculateSpace();
  }
 
  /**
   * MAX METAL pg 14
   * Armor: Maximum armor SP total is 1/2 of SDP total. Each 10% (or part thereof)
   * of a vehicle's SDP in SP lowers vehicle's speed by 10% (this lowers an
   * Osprey's or helicopter's speed by 20% per 10% SP).
   * Armor costs are determined
   * by the amount of armore desired - vehicle armored from 1-20 SP pay 500eb
   * per SP. Vehicle arrmored from 21-40 SP pay 1,000eb per SP. Vehicles with 41-60
   * SP pay 5,000eb per SP.
   * Vehicles with over 60 SP armor pay 7,000eb per SP. For instance, 50 SP
   * of vehicle armor costs (50x5,000) = 250,000eb. Ospreys and airplanes can have
   * a maximum SP armor of 1/4 their SDP. Airships can't be armored, an airship
   * gondala can, but the gasbag can't; and the gasbag is hit 90% of the time,
   * unless specifically aiming for the gondala.
   * A vehicle without armor actually weighs half the listed mass. The mass
   * calculations above are retained for purposes of mounting pod weapons and
   * collisions.
   * @param {number} value
   * @memberof MaxmetalService
   */
  changeSP(value: number) {
    // only change if the value is within range.
    if (this.currVehicle.type.name.toUpperCase() !== 'AIRSHIP') {
      this.currVehicle.sp.curr += value;
      // sp only affects mass if it is 0SP.
      if ( this.currVehicle.sp.curr > 0 ) {
        this.calculateMass();
      }
      // calculate the cost
      if ( this.currVehicle.sp.curr  > 0 && this.currVehicle.sp.curr  < 21 ) {
        this.currVehicle.sp.cost = this.currVehicle.sp.curr  * 500;
      } else if (this.currVehicle.sp.curr  > 20 && this.currVehicle.sp.curr  < 41) {
        this.currVehicle.sp.cost = this.currVehicle.sp.curr  * 1000;
      } else if (this.currVehicle.sp.curr  > 40 && this.currVehicle.sp.curr  < 61) {
        this.currVehicle.sp.cost = this.currVehicle.sp.curr  * 5000;
      } else if (this.currVehicle.sp.curr  > 60 ) {
        this.currVehicle.sp.cost = this.currVehicle.sp.curr  * 7000;
      }
      // adjust topspeed
      let spdFactor = Math.floor((this.currVehicle.sp.curr / this.currVehicle.sdp.curr) * 10) / 10;
      if ( this.currVehicle.type.name.toUpperCase() === 'OSPREY' ||
        this.currVehicle.type.name.toUpperCase().indexOf('HELICOPTER') > -1) {
          // each 10% of SDP in SP
          spdFactor = spdFactor * 2;
      }
      this.currVehicle.sp.spdMod = spdFactor;
 
      this.calculateSpeed();
      this.calculateSpace();
      this.calculateCost();
    }
  }

  changeHandling(value: number) {

  /**
   * MAX METAL pg. 14
   * Boosted Acceleration: Each 10% acceleration added to a vehicle's
   * acceleration (minimum boost allowed is 5mph) costs 5% of the vehicle's
   * base SDP cost. Maximum boost 100% acceleration.
   * @param {number} value
   * @memberof MaxmetalService

  changeAcceleration(value: number) {
    this.currVehicle.accelerate.curr += value;
    const boost = this.currVehicle.accelerate.curr - this.currVehicle.accelerate.base;
    console.log(`boost: ${boost}`);
    if ( value > 0 && boost < 5 ) {
      this.currVehicle.accelerate.curr = this.currVehicle.accelerate.base + 5;
    }
    if ( value < 0 && boost < 5 ) {
      this.currVehicle.accelerate.curr = this.currVehicle.accelerate.base;
    }
    // calculate sdp cost
    const factor = Math.floor((( this.currVehicle.accelerate.curr / this.currVehicle.accelerate.base ) % 1) * 10);
    this.currVehicle.accelerate.cost = 1 + (0.05 * factor);
    this.calculateCost();
  }
 
 
  /**
   * Heavy-Duty Brakes: Each 25% of deceleration added to a vehicle's deceleration

   * vehicles (hovers count as an air vehicle for this option).

    this.calculateCost();
  }

  calculateAdjSDP() {
    let min = 0;
    let max = 0;
    if (this.currVehicle.sdp.base <= this.currVehicle.sdp.min) {
      min = Math.round(this.currVehicle.type.sdp.min * 0.5) - this.currVehicle.type.sdp.min;
    }
    if (this.currVehicle.sdp.base >= this.currVehicle.sdp.max) {
      max = Math.round(this.currVehicle.type.sdp.max * 1.25) - this.currVehicle.type.sdp.max;
    }
    this.currVehicle.sdpAdj.min = min;
    this.currVehicle.sdpAdj.max = max;
    this.currVehicle.sdpAdj.base = 0;
    if (this.currVehicle.sdpAdj.curr && this.currVehicle.sdpAdj.curr !== 0 ) {
      this.currVehicle.sdp.curr -= this.currVehicle.sdpAdj.curr;
    }
    this.currVehicle.sdpAdj.curr = 0;
  }
 
 /**
   * MAX METAL pg 14
   * Ospreys and airplanes can have
   * a maximum SP armor of 1/4 their SDP. Airships can't be armored, an airship
   * gondala can, but the gasbag can't; and the gasbag is hit 90% of the time,
   * unless specifically aiming for the gondala.
   * @memberof MaxmetalService
   */
  calculateSP() {
    // Airships can't be armored.
    if (this.currVehicle.type.name.toUpperCase() === 'AIRSHIP') {
      this.currVehicle.sp.max = 0;
    } else if ( this.currVehicle.type.name.toUpperCase() === 'OSPREY' ||
      this.currVehicle.type.name.toUpperCase().indexOf('PLANE') > -1 ) {
        this.currVehicle.sp.max = Math.round(this.currVehicle.sdp.curr * 0.25);
    } else {
      this.currVehicle.sp.max = Math.round(this.currVehicle.sdp.curr * 0.5);
    }
  }
 
  /**
   * calculates the topspeed object withint he currVeicle.
   * topspeed is based off the speed of the vehicle type.
   * This will be adjusted by SP value as well.
   * @memberof MaxmetalService
   */
  calculateSpeed() {
    // adjust the speed based on SP
    const modifier =   1 - this.currVehicle.sp.spdMod;
    // reset the base and max topspeed
    this.currVehicle.topspeed.base = Math.floor(this.currVehicle.type.speed * modifier);
    this.currVehicle.topspeed.max = this.currVehicle.topspeed.base * 2;
    this.currVehicle.topspeed.min = Math.round(this.currVehicle.topspeed.base * 0.1);
    this.currVehicle.topspeed.curr = (this.currVehicle.topspeed.curr) ? this.currVehicle.topspeed.curr : this.currVehicle.type.speed;
    if ( this.currVehicle.topspeed.curr > this.currVehicle.topspeed.max) {
      this.currVehicle.topspeed.curr = this.currVehicle.topspeed.max;
    }
    // caculate the eb cost
    if ( this.currVehicle.topspeed.curr > this.currVehicle.topspeed.base) {
      // increase SDP by 25% per 10% of speed
      this.currVehicle.topspeed.cost = 1 + ( 0.25 *
        (Math.floor((this.currVehicle.topspeed.curr / this.currVehicle.topspeed.base) * 10) % 10) );
    } else if ( this.currVehicle.topspeed.curr < this.currVehicle.topspeed.base) {
      // decrease SDP by 10% per 10% of speed
      this.currVehicle.topspeed.cost = 0.1 * (
        (Math.ceil((this.currVehicle.topspeed.curr / this.currVehicle.topspeed.base) * 10) % 10));
      if ( this.currVehicle.topspeed.cost <= 0 ) {
        this.currVehicle.topspeed.cost = 1;
      }
    } else {
      this.currVehicle.topspeed.cost = 1;
    }
    console.log(`topspeed cost: ${this.currVehicle.topspeed.cost}`);
  }
 
  /**
   * calculate the total cost of the vehicle. The cost values of SDP and armor
   * reside in those methods. Other factors are set on the different object as
   * well.
   * @memberof MaxmetalService
   */
  calculateCost() {
    let cost = 0;
    let multiplier = 1;
    cost = this.currVehicle.sdp.curr * this.currVehicle.type.sdp.eb;
    // extra structure cost double for each point.
    if ( this.currVehicle.sdpAdj.curr > 0 ) {
      cost += (this.currVehicle.sdpAdj.curr * this.currVehicle.type.sdp.eb);
    }
    multiplier = multiplier * this.currVehicle.manuever.cost;
    multiplier = multiplier * this.currVehicle.topspeed.cost;
    multiplier = multiplier * this.currVehicle.accelerate.cost;
    // off road is a flat cost of 15%
    if (this.currVehicle.offRoad) {
      multiplier = multiplier * 1.15;
    }
    cost = cost * multiplier;
    // armor isn't a multiplier of the sdp cost.
    cost += this.currVehicle.sp.cost;
    this.currVehicle.cost = Math.ceil(cost);
  }
 
  calculateMass() {
    this.currVehicle.mass.base = Math.round(
      (this.currVehicle.type.mass.wt * (this.currVehicle.sdp.curr / this.currVehicle.type.mass.sdp)) * 100) / 100;
    // if the vehicle has no armor than it is half it's wait.
    if (this.currVehicle.sp.curr < 1 ) {
      this.currVehicle.mass.base = this.currVehicle.mass.base / 2;
    }
    this.currVehicle.mass.curr = this.currVehicle.mass.base;
    this.currVehicle.mass.value = this.currVehicle.mass.base + ' ' + this.currVehicle.type.mass.unit;
  }
 
  calculateSpace() {

    if (this.currVehicle.type.sdp.perSpace > 0) {
       spaces = Math.floor(this.currVehicle.sdp.curr / this.currVehicle.type.sdp.perSpace);
    }
    this.currVehicle.spaces.base = spaces;
    // lose 5% of spaces for each 10% of speed over base
    let spdCost = 0;
    if (this.currVehicle.topspeed.curr > this.currVehicle.topspeed.base ) {
      const spdFactor = Math.floor(((this.currVehicle.topspeed.curr / this.currVehicle.topspeed.base) - 1 ) * 10);
      spdCost = -(spdFactor * 0.05 * spaces);
    }
    if (this.currVehicle.topspeed.curr < this.currVehicle.topspeed.base ) {
      const spdFactor = Math.ceil((1 - (this.currVehicle.topspeed.curr / this.currVehicle.topspeed.base)) * 10 );
      spdCost = spdFactor * 0.1 * spaces;
    }
    // range modifies the space available.
    const rngFactor = (this.currVehicle.range.curr / this.currVehicle.range.base) * 100;
    if (rngFactor < 100 ) {
      // add 10% of space for each 33% decrease in range.
    } else {
      // lose 10% of space for each 33% increase in range.
    }
 
    spaces = Math.floor((spaces + spdCost) * 10 ) / 10;
    if (spaces < this.currVehicle.type.spaces.min) {
     spaces = this.currVehicle.type.spaces.min;
    }
    if (spaces > this.currVehicle.type.spaces.max) {
     spaces = this.currVehicle.type.spaces.max;
    }
    this.currVehicle.spaces.curr = spaces;
  }
